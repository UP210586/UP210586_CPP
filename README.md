int main()
{
   int edad;
   cout << "Dame tu edad ";
   cin >> edad;

   if (edad >= 1 && edad <= 150)
   {
      if (edad <= 30)
      {
         cout << "Primera Edad" << endl;
      }
      else if (edad >= 31 && edad <= 60)
      {
         cout << "Segunda Edad" << endl;
      }
      else if (edad >= 61 && edad <= 90)
      {
         cout << "Tercera Edad" << endl;
      }
      else
      {
         cout << "Horas Extra" << endl;
      }
   }
   else
   {
      cout << "El rango es invalido" << endl;
   }
   getchar();
   return 0;
}
